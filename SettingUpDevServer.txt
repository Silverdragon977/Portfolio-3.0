        then npx vite build


To create a Model and Controller
    php artisan make:model NameOfController -mcr\
    // To update database/migrations/table.php
    php artisan migrate
    // To rollback mistake
    php artisan migrate:rollback
    // Target specific file? 
    --path=database/migrations/NameOfModel.php
    // Delete and re-run all Models?
    php artisan migrate:refresh
